      <RealTimeChart data={metrics.chartData} />
      <LiveActivityFeed events={metrics.events} />
    </div>
  );
};

  const spring = useSpring({ number: value, from: { number: 0 } });
  
  return (

      {spring.number.to(n => Math.floor(n))}
    </animated.div>
  );
};
